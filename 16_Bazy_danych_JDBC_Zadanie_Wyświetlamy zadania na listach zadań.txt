SELECT I.SUMMARY, I.DESCRIPTION, L.NAME AS 'ISSUES LIST NAME'
FROM issues I JOIN issueslist L on L.ID = I.ISSUESLIST_ID;

SELECT I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM issues I JOIN USERS U on U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM USERS U JOIN issues i on U.ID = i.USER_ID_ASSIGNEDTO
GROUP BY U.ID
HAVING (COUNT(*) > 0);
